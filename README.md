<!-- interface -->

interface => যেটা আগেই বলে দেওয়া কোথায় কি আসবে ডেস্ক্রিপশন আসবে কি না।

যেমনঃ একটা কীবোর্ড দারাজ বিক্রি করবে সেটার জন্য তার নাম, দাম রাখা বাঁধতা মূলক সেটা কেই interface বোঝাই

উদহারণ =>

const tel: Product = {

nam: "mouse",

dam: 1500,
};

কোথায় কি আসবে সেটা আগে থেকেই নির্দিষ্ট করে বলে দেওয়া।

<!-- Type -->

Type => কীবোর্ড ব্লুটুথ হবে নাকি কীবোর্ড ব্লুটুথ হবে না সেটা আগে থেকেই বলে দেয়

যেমনঃ কীবোর্ড কেনার পর কিভাবে টাকা দিবো বিকাশ, নগদ, নাকি হাতে ক্যাশ টাকা দিবো আগেই বলতে হবে। তবে এর বাইরে দেওয়া যাবে না।

উদহারণ =>

type PaymentMethod = "bKash" | "Cash";

let taka2: PaymentMethod = "Nagad"; => টাকা পাঠাতে পারবে না কারণ type এ এইটা বলা নাই

<!-- any -->

any => সবকিছুই চলবে

যেমনঃ দারাজে প্রায় সব রকম জিনিস পাওয়া যায়, কোনো নির্দিষ্ট বাঁধা নাই।

উদহারণ =>

let item: any;

item = "Mouse"; => মাউস (string)

item = 1000; => ১টা মাউস এর দাম (number)

item = {

name: 'mouse',

quantity: 2,

description: '2.4GHz, Bluetooth, Wired'

}; => ডেস্ক্রিপশন (object)

item = isAvailable; => আছে বা নাই (boolean)

কোনো প্রপার্টি বানান বা কোনো কিছু ভুল লিখলে ধরবে না স্ট্রিং দরকার। কিন্তু নাম্বার দিলে কোনো কিছু বলবে না কোনো

<!-- unknown -->

unknown => সব কিছু ব্যবহার করা যাবে কিন্তু ব্যবহার করার আগে দেখে করতে হবে।

যেমনঃ দারাজ থেকে মাউস টা কিনলাম সেটা ডেলিভারি মেন্ থেকে নেওয়ার সময় সেটা চাক করে নেওয়া।

ডেসক্রিপশন যেটা বলছে সেটাই দিচ্ছে কি না।

উদহারণ =>

let mouses: unknown = "mouse"; => যেকোনো কিছু পাঠাতে পারবে কোনো কিছু বলবে না || string, number, object, boolean

if (typeof mouses === "string") { => এখানে তাকে দেখতে হবে কি পাঠালো দারাজ থেকে যা অর্ডার দেওয়া হয়েছে সেটা পাঠাইছে কি না

console.log("Item:", mouses.toUpperCase());
}

unknown = নিরাপদ any থেকে কারণ সে দেখে নেই যা পাঠায় তাই নেই না।

<!-- never -->

never => যেটা একদম অসম্ভব জিনিস যেটা কখনো হবে না

যেমনঃ একটি মাউস এর দাম -১০০০০ টাকা কখনই হবে না। এমন কিছু কে never বলা হয়
